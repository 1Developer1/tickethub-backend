async execute<T>(fn: () => Promise<T>): Promise<T> { if (this.state === 'OPEN') { if (Date.now() - this.lastFailureTime > this.cooldownMs) this.state = 'HALF_OPEN'; else throw new ExternalServiceError(); // fail fast! } try { const result = await fn(); this.onSuccess(); // → CLOSED return result; } catch (e) { this.onFailure(); // 5 hata → OPEN throw e; } }
